# QMK Leader key mapping

- non-grouped
    - A + S => Win + S
    - C + C => copy
    - C + V => paste
- V -- Vim
    - C => change inside ''
- S -- Shell
    - S => run UTF8 decode script
    - Q => backup json file
    - D => delete all json files in folder
    - B => go to backup folder
    - C => git add & commit with message
- M -- Mail
    <!-- Outlook -->
    - D => mark as read and delete
    - F => mark as read and flag with no date
    - L => assign label
    - M => mark flag done and move mail
    <!-- often used text in mails -->
    - R => machines uploaded
    - F + G => Mit freundlichen Grüßen
    - S => Seehr geehrte Damen und Herren,
    - S + E => Seehr geehrte Frau
    - S + R => Seehr geehrter Herr
- N -- Names
    - A => Ábel
    - K => Katharina
    - M => Marek
    - V => Valentin
- T -- Text
    - T => Thank you
    - T + X => Thanks
